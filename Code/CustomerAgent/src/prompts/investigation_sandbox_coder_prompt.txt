```python
import json, pandas as pd
from azure.core.credentials import AccessToken, TokenCredential
from azure.storage.filedatalake import DataLakeServiceClient

class _StaticTokenCredential(TokenCredential):
    def __init__(self, token, expires_on):
        self._t = AccessToken(token, int(expires_on))
    def get_token(self, *scopes, **kw):
        return self._t

_FS = DataLakeServiceClient(
    account_url=f"https://{ADLS_ACCOUNT}.dfs.core.windows.net",
    credential=_StaticTokenCredential(ADLS_TOKEN, ADLS_TOKEN_EXPIRES_ON),
).get_file_system_client(ADLS_FILESYSTEM)

def adls_exists(path: str) -> bool:
    try:
        _FS.get_file_client(path.lstrip("/")).get_file_properties()
        return True
    except Exception:
        return False

def adls_read_text(path: str) -> str:
    return _FS.get_file_client(path.lstrip("/")).download_file().readall().decode("utf-8")

def adls_write_text(path: str, content: str) -> None:
    fc = _FS.get_file_client(path.lstrip("/"))
    fc.upload_data(content.encode("utf-8"), overwrite=True)

def adls_list(prefix: str) -> list:
    try:
        return [p.name for p in _FS.get_paths(path=prefix.lstrip("/"), recursive=True) if not p.is_directory]
    except Exception:
        return []
```

JSON SERIALIZATION
═══════════════════════════════════════════════════════════════════

pandas/numpy types are NOT JSON-serializable. ALWAYS define these helpers
at the TOP of your script — note that `_safe` and `_json_dumps` are also

```python
import numpy as np, pandas as pd

def _safe(obj):
    if isinstance(obj, (np.integer,)): return int(obj)
    if isinstance(obj, (np.floating,)): return float(obj)
    if isinstance(obj, (np.bool_,)): return bool(obj)
    if isinstance(obj, (np.ndarray,)): return obj.tolist()
    if isinstance(obj, (pd.Timestamp,)): return obj.isoformat()
    if pd.isna(obj): return None
    return str(obj)

def _json_dumps(obj, **kw):
    kw.setdefault("default", _safe); kw.setdefault("indent", 2)
    try: return json.dumps(obj, **kw)
    except ValueError:
        def _bc(o, s=None):
            if s is None: s = set()
            i = id(o)
            if isinstance(o, dict):
                if i in s: return "{...}"
                s.add(i); return {k: _bc(v, s) for k, v in o.items()}
            if isinstance(o, (list, tuple)):
                if i in s: return "[...]"
                s.add(i); return [_bc(v, s) for v in o]
            return o
        return json.dumps(_bc(obj), **kw)
```

Use `_json_dumps(...)` for ALL JSON serialization. NEVER call `json.dumps()`
directly.
